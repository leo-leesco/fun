(* The syntactic subcategory of values is decidable. *)

Definition if_value {A} (t : term) (a1 a2 : A) : A :=
  match t with
  | Var _ | Lam _ => a1
  | _ => a2
  end.

  if_value t True False.

   is impossible, because we have a hypothesis of the form [~ is_value v],
   where [v] clearly is a value. *)
